import requests # Trello API key and token api_key = 'your_trello_api_key' api_token = 'your_trello_api_token' # Board and list ID board_id = 'your_board_id' list_id = 'your_list_id' # Trello API endpoint for cards in a list url = f'https://api.trello.com/1/lists/{list_id}/cards' # Parameters including API key and token params = { 'key': api_key, 'token': api_token } # Make a GET request to retrieve cards response = requests.get(url, params=params) # Check if the request was successful if response.status_code == 200: # Print or process the cards cards = response.json() for card in cards: print(card['name']) else: print(f"Error: {response.status_code}")
